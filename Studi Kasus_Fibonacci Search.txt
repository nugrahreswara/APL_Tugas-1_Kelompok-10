#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int fibonacciSearch(vector<int> A, int x, int n) {
    int fibMMm2 = 0;
    int fibMMm1 = 1;
    int fibM = fibMMm2 + fibMMm1;

    while (fibM < n) {
        fibMMm2 = fibMMm1;
        fibMMm1 = fibM;
        fibM = fibMMm2 + fibMMm1;
    }

    int offset = -1;

    while (fibM > 1) {
        int i = min(offset + fibMMm2, n - 1);

        if (A[i] < x) {
            fibM = fibMMm1;
            fibMMm1 = fibMMm2;
            fibMMm2 = fibM - fibMMm1;
            offset = i;
        }

        else if (A[i] > x) {
            fibM = fibMMm2;
            fibMMm1 = fibMMm1 - fibMMm2;
            fibMMm2 = fibM - fibMMm1;
        }

        else {
            return i;
        }
    }

    if (fibMMm1 == 1 && A[offset + 1] == x) {
        return offset + 1;
    }

    return -1;
}

int main() {
    vector<int> hargaBarang = {10, 22, 35, 40, 45, 50, 70, 80, 82, 90, 100};
    int n = hargaBarang.size();
    int target = 70;

    cout << "Daftar Harga Barang: ";
    for (int h : hargaBarang) cout << h << " ";
    cout << "\nTarget harga yang dicari: " << target << endl;

    int hasil = fibonacciSearch(hargaBarang, target, n);

    if (hasil != -1) {
        cout << "\nHarga " << target << " ditemukan pada indeks ke-" << hasil << endl;
    } else {
        cout << "\nHarga " << target << " tidak ditemukan dalam daftar." << endl;
    }

    return 0;
}
