#include <iostream>
using namespace std;

void shellSort(int A[], int n) {
    for (int gap = n / 2; gap > 0; gap /= 2) {
        for (int i = gap; i < n; i++) {
            int temp = A[i];
            int j = i;

            while (j >= gap && A[j - gap] > temp) {
                A[j] = A[j - gap];
                j -= gap;
            }
            
            A[j] = temp;
        }
    }
}

void printArray(int A[], int n) {
    for (int i = 0; i < n; i++) {
        cout << A[i] << " ";
    }
    cout << endl;
}

int main() {
    int A[] = {9, 8, 3, 7, 5, 6, 4};
    int n = sizeof(A) / sizeof(A[0]);
    cout << "Array sebelum diurutkan: ";
    printArray(A, n);
    shellSort(A, n);
    cout << "Array setelah diurutkan (Shell Sort): ";
    printArray(A, n);
    return 0;
}
